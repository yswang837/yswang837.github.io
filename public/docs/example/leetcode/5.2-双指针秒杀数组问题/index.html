    双指针秒杀数组问题 #

      在数组中，下标就可以认为是指针。双指针主要分为：左右指针，快慢指针。

      左右指针：从数组的两个端点向中间移动，比如说二分查找及其相关的变种题；从字符串中间向两边移动，如最长回文子串。

      快慢指针：都从数组的起点开始，快指针负责探路，慢指针跟快指针配合完成题目要求，比如说去掉有序数组的重复元素，并返回新数组长度。

    例：二分查找 #

      二分查找是典型的左右指针，针对的场景是一切有序数组，时间复杂度为O(logN)。

    func binarySearch(nums []int, target int) int {
	if len(nums) == 0 {
		return -1
	}
	left, middle, right := 0, 0, len(nums)-1
	for left <= right {
		middle = (left + right) / 2
		if nums[middle] == target {
			return middle
		} else if nums[middle] > target {
			right = middle - 1
		} else {
			left = middle + 1
		}
	}
	return -2
}

    26.删除有序数组中的重复项 #

    传送门

    func removeDuplicates(nums []int) int {
    if len(nums) == 0 {
        return 0
    }
    slow, fast := 0, 0
    for fast < len(nums) {
        if nums[slow] == nums[fast] {
            fast++
        }else {
            slow++
            nums[slow] = nums[fast]
        }
    }
    return slow+1
}

    167.两数之和 II - 输入有序数组 #

    传送门

    func twoSum(numbers []int, target int) []int {
    if len(numbers) == 0 {
        return numbers
    }
    left, right := 0, len(numbers) - 1
    sum := 0
    for left <= right {
        sum = numbers[left] + numbers[right]
        if sum == target {
            return []int{left+1,right+1}
        }else if sum > target {
            right--
        }else {
            left++
        }
    }
    return []int{-1,-1}
}

    283.移动零 #

    传送门

    func moveZeroes(nums []int) {
    if len(nums) == 0 {
        return
    }
    slow, fast := 0, 0
    for fast < len(nums) {
        if nums[fast] != 0 {
            nums[slow], nums[fast] = nums[fast], nums[slow]
            slow++
        }
        fast++
    }
}

    344.反转字符串 #

      反转数组一个意思。

    传送门

    func reverseString(s []byte)  {
    if len(s) == 0 {
        return
    }
    left, right := 0, len(s) - 1
    for left <= right {
        s[left], s[right] = s[right], s[left]
        left++
        right--
    }
    return
}

    5.最长回文子串 #

      首先明确一下，回文串就是正着读和反着读都一样的字符串，判断回文串很简单，用左右指针相向而行就行了，而且不需要考虑是奇数还是偶数。但寻找最长回文子串，需要考虑奇数还是偶数，同样采取左右指针，但这个左右指针是相背而行，从中间向两端搜索。

    传送门

    func longestPalindrome(s string) string {
    res := ""
    for i := 0; i < len(s); i++ {
        // 以 s[i] 为中心的最长回文子串
        s1 := palindrome(s, i, i)
        // 以 s[i] 和 s[i+1] 为中心的最长回文子串
        s2 := palindrome(s, i, i + 1)
        res = maxLenOfString(res,s1)
        res = maxLenOfString(res,s2)
    }
    return res
}

func palindrome(s string, l, r int) string {
		if l > r {
			return ""
		}
		for l >= 0 && r < len(s) && s[l] == s[r] {
			l--
			r++
		}
		return s[l+1 : r] // 本来r也要减减，但由于是左闭右开的区间，所以刚好不用
}

func maxLenOfString(s1, s2 string) string{
    if len(s1) >= len(s2) {
        return s1
    }
    return s2
}

    179.查找总价格为目标值的两个商品(剑指Offer) #

    传送门

    func twoSum(price []int, target int) []int {
    if len(price) == 0 {
        return nil
    }
    left, right := 0, len(price) - 1
    sum := 0
    leftVal,rightVal := 0, 0
    for left <= right {
        leftVal,rightVal = price[left], price[right]
        sum = leftVal + rightVal
        if sum == target {
            return []int{leftVal,rightVal}
        }else if sum > target {
            right--
        }else {
            left++
        }
    }
    return nil
}
