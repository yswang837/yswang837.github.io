func (p *problems) randProblem(t int) string {
	if t < 0 || t > 17 {
		return ""
	}
	if t == 0 {
		allProblem := []string{}
		for _, val := range p.kind {
			allProblem = append(allProblem, val.numbers...)
		}
		return allProblem[rand.Intn(100)]
	} else {
		numbers := p.kind[t-1].numbers
		return numbers[rand.Intn(len(numbers))]
	}

}
